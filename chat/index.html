script Loader

    variable Script
    variable Credentials
    variable ChatConfig
    variable Broker
    variable MqttUsername
    variable MqttPassword
    variable SystemID
    variable MyID
    variable ReceivedMessage
    variable WaitCount
    topic MyTopic
    topic ServerTopic
    
!    debug step

    ! Get MQTT credentials
    log `GET ../credentials.json`
    rest get Credentials from `../credentials.json`
    or begin
    	log `That failed, so GET ../credentials.php`
        rest get Credentials from `../credentials.php` or alert `Cannot load credentials`
        go to GotCredentials
    end

GotCredentials:
!    log Credentials
    put element `broker` of Credentials into Broker
    put element `username` of Credentials into MqttUsername
    put element `password` of Credentials into MqttPassword

    ! Get chat config
    rest get ChatConfig from `chat-config.json`
    put element `id` of ChatConfig into SystemID
    put SystemID cat `/chat` into SystemID
    log `System ID is ` cat SystemID

    ! Connect to MQTT
    put `Loader-` cat random 999999 into MyID
    init ServerTopic
        name SystemID
        qos 1
    init MyTopic
        name MyID
        qos 1
    mqtt
        token MqttUsername MqttPassword
        id MyID
        broker Broker
        port 443
        subscribe MyTopic

    on mqtt connect
    begin
        go to RequestScript
    end

    on mqtt message
    begin
        put the mqtt message into ReceivedMessage
    end

    log `Waiting for MQTT to connect`
    stop

RequestScript:
    log `Request the main script`
    send to ServerTopic
        sender MyTopic
        action `get-file`
        message `chat-main.ecs`
    put 0 into WaitCount
WaitLoop:
    if ReceivedMessage is not empty go to GotScript
    add 1 to WaitCount
    if WaitCount is greater than 200
    begin
        alert `Server did not respond`
        stop
    end
    wait 10 ticks
    go to WaitLoop

GotScript:
    put ReceivedMessage into Script
    run Script
